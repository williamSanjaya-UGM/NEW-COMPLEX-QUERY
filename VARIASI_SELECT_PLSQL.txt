create or replace PROCEDURE USP_TEST_NUMBER(
    P_PARAM1 NUMERIC,
    P_PARAM2 NUMERIC,
    P_CUR OUT SYS_REFCURSOR,
    P_CUR1 OUT SYS_REFCURSOR
)AS
BEGIN
    OPEN P_CUR FOR
    SELECT NUMBER_1
    FROM TESTING_TBL
    WHERE
--        NVL(NUMBER_1,0) BETWEEN NVL(P_PARAM1,NVL(NUMBER_1,0)) AND NVL(P_PARAM2,NVL(NUMBER_1,0)); -- IF ALSO WANT TO SHOW NULL VALUE
    NVL(NUMBER_1,NULL) BETWEEN NVL(P_PARAM1,NVL(NUMBER_1,NULL)) AND NVL(P_PARAM2,NVL(NUMBER_1,NULL)) -- IF DON'T WANT TO SHOW NULL VALUE
    AND (CASE P_PARAM2
        WHEN NULL THEN 1
        ELSE 0
        END)=1;

END;
